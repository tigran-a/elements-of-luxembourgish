\section{Goen vs Ginn}

\begin{itemize}
    \item \textbf{Ech ginn nach verréckt.} \\
    (I am going crazy. / Je deviens fou.)
    \item \textbf{Ech ginn dir e gudde Rot.} \\
    (I give you good advice. / Je te donne un bon conseil.)
    \item \textbf{Ech ginn an d’Stad.} \\
    (I go to the city. / Je vais en ville.)
\end{itemize}

\subsection*{Ausdr\"{e}ck mam Wuert "D\"{e}sch" / Expressions with "Table" / Expressions avec "Table"}
\begin{itemize}
    \item \textbf{um D\"{e}sch setzen}
    \begin{itemize}
        \item Lit: to sit at the table / \^{e}tre assis \`a table
    \end{itemize}
    \item \textbf{Iwwer D\"{e}sch g\"e}t net geschwat
    \begin{itemize}
        \item Lit: One does not speak across the table. / On ne parle pas \`a travers la table.
        \item Fig: One does not speak while eating. / On ne parle pas pendant qu'on mange. \\
        \textit{(iwwerem Iesse g\"e}t net geschwat)
    \end{itemize}
    \item \textbf{Meng Propositioun ass \"unner den D\"{e}sch gefall}
    \begin{itemize}
        \item Lit: My proposal fell under the table. / Ma proposition est tomb\'{e}e sous la table.
        \item Fig: My proposal was not retained/accepted. / Ma proposition n'a pas \'{e}t\'{e} retenue. \\
        \textit{(meng Propositioun ass net zer\'{e}ckbehale ginn)}
    \end{itemize}
\end{itemize}
