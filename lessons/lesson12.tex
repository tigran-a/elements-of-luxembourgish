\section{Rods wien ech sinn / Guess who I am / Devine qui je suis}

Dëst ass e Spill wou eng Persoun un eng berüümt Persoun denkt, an déi aner musse froen fir erauszefannen wien et ass.
\textit{(This is a game where one person thinks of a famous person, and the others have to ask questions to find out who it is. / C'est un jeu où une personne pense à une personne célèbre et les autres doivent poser des questions pour découvrir de qui il s'agit.)}

\begin{enumerate}
    \item \textbf{Ass et e Mann?} \\ \textit{(Is it a man? / Est-ce un homme ?)}
    \item \textbf{Ass et eng Fra?} \\ \textit{(Is it a woman? / Est-ce une femme ?)}
    \item \textbf{Lieft déi Persoun nach?} \\ \textit{(Is that person still alive? / Cette personne est-elle encore en vie ?)}
    \item \textbf{Ass déi Persoun doud?} \\ \textit{(Is that person dead? / Cette personne est-elle morte ?)}
    \item \textbf{Ass si scho laang gestuerwen?} \\ \textit{(Did she/he die a long time ago? / Est-elle/il décédé(e) il y a longtemps ?)}
    \item \textbf{Ass et eng richteg Persoun?} \\ \textit{(Is it a real person? / Est-ce une personne réelle ?)}
    \item \textbf{Ass et eng fiktiv Figur?} \\ \textit{(Is it a fictional character? / Est-ce un personnage fictif ?)}
    \item \textbf{Ass si weltberüümt?} \\ \textit{(Is she/he world famous? / Est-elle/il mondialement célèbre ?)}
    \item \textbf{Kënnt si aus Lëtzebuerg?} \\ \textit{(Does she/he come from Luxembourg? / Vient-elle/il du Luxembourg ?)}
    \item \textbf{Ass si Europäer / Europäerin?} \\ \textit{(Is she/he European? / Est-elle/il européen(ne) ?)}
    \item \textbf{Ass si Amerikaner / Amerikanerin?} \\ \textit{(Is she/he American? / Est-elle/il américain(e) ?)}
    \item \textbf{Ass si bestued?} \\ \textit{(Is she/he married? / Est-elle/il marié(e) ?)}
    \item \textbf{Huet si Kanner?} \\ \textit{(Does she/he have children? / A-t-elle/il des enfants ?)}
    \item \textbf{Ass si jonk?} \\ \textit{(Is she/he young? / Est-elle/il jeune ?)}
    \item \textbf{Ass si al?} \\ \textit{(Is she/he old? / Est-elle/il âgé(e) ?)}
    \item \textbf{Huet si blond Hoer?} \\ \textit{(Does she/he have blonde hair? / A-t-elle/il les cheveux blonds ?)}
    \item \textbf{Huet si schwaarz Hoer?} \\ \textit{(Does she/he have black hair? / A-t-elle/il les cheveux noirs ?)}
    \item \textbf{Huet si laang Hoer?} \\ \textit{(Does she/he have long hair? / A-t-elle/il les cheveux longs ?)}
    \item \textbf{Huet si kuerz Hoer?} \\ \textit{(Does she/he have short hair? / A-t-elle/il les cheveux courts ?)}
    \item \textbf{Dréit si e Brëll?} \\ \textit{(Does she/he wear glasses? / Porte-t-elle/il des lunettes ?)}
    \item \textbf{Huet hien e Baart?} \\ \textit{(Does he have a beard? / A-t-il une barbe ?)}
    \item \textbf{Huet hien e Schnurres?} \\ \textit{(Does he have a mustache? / A-t-il une moustache ?)}
    \item \textbf{Ass si e Politiker / eng Politikerin?} \\ \textit{(Is she/he a politician? / Est-elle/il politicien(ne) ?)}
    \item \textbf{Ass si e Sänger / eng Sängerin?} \\ \textit{(Is she/he a singer? / Est-elle/il chanteur/chanteuse ?)}
    \item \textbf{Ass si e Schauspiller / eng Schauspillerin?} \\ \textit{(Is she/he an actor/actress? / Est-elle/il acteur/actrice ?)}
    \item \textbf{Ass si e Sportler / eng Sportlerin?} \\ \textit{(Is she/he an athlete? / Est-elle/il sportif/sportive ?)}
    \item \textbf{Spillt si Fussball?} \\ \textit{(Does she/he play football? / Joue-t-elle/il au football ?)}
    \item \textbf{Schafft si op der Tëlee?} \\ \textit{(Does she/he work on TV? / Travaille-t-elle/il à la télé ?)}
    \item \textbf{Ass si e Wëssenschaftler?} \\ \textit{(Is she/he a scientist? / Est-elle/il scientifique ?)}
    \item \textbf{Ass si e Kënschtler?} \\ \textit{(Is she/he an artist? / Est-elle/il artiste ?)}
    \item \textbf{Ass si e Schrëftsteller?} \\ \textit{(Is she/he a writer? / Est-elle/il écrivain ?)}
    \item \textbf{Huet si e Buch geschriwwen?} \\ \textit{(Did she/he write a book? / A-t-elle/il écrit un livre ?)}
    \item \textbf{Ass si an engem Film?} \\ \textit{(Is she/he in a movie? / Est-elle/il dans un film ?)}
    \item \textbf{Ass si räich?} \\ \textit{(Is she/he rich? / Est-elle/il riche ?)}
    \item \textbf{Ass si e Kinnek oder eng Kinnigin?} \\ \textit{(Is she/he a king or a queen? / Est-elle/il roi ou reine ?)}
    \item \textbf{Ass si historesch wichteg?} \\ \textit{(Is she/he historically important? / Est-elle/il historiquement important(e) ?)}
    \item \textbf{Huet si eppes erfond?} \\ \textit{(Did she/he invent something? / A-t-elle/il inventé quelque chose ?)}
    \item \textbf{Huet si e Präis gewonnen?} \\ \textit{(Did she/he win an award? / A-t-elle/il gagné un prix ?)}
    \item \textbf{Sinn d'Leit frou mat hir?} \\ \textit{(Do people like her/him? / Les gens l'aiment-ils ?)}
    \item \textbf{Ass si eng béis Persoun?} \\ \textit{(Is she/he a bad person? / Est-ce une mauvaise personne ?)}
\end{enumerate}

\section{Wieder Sätz / Weather Phrases / Phrases Météo}

\begin{itemize}
    \item \textbf{Haut ass et waarm.} \\
    \textit{(Today it is warm. / Aujourd'hui, il fait chaud.)}
    
    \item \textbf{Haut ass et ganz waarm.} / \textbf{Haut ass et déck waarm.} \\
    \textit{(Today it is very warm. / Aujourd'hui, il fait très chaud.)} \\
    \textit{Note: "déck" is more colloquial/informal.}
    
    \item \textbf{Haut ass et ze waarm.} \\
    \textit{(Today is too warm. / Aujourd'hui, il fait trop chaud.)}
    
    \item \textbf{Oft ass et hei ganz kal.} \\
    \textit{(Often it is very cold here. / Souvent, il fait très froid ici.)}
\end{itemize}
